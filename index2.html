SQL-запросы

SELECT, JOIN, GROUP BY, подзапросы и отчеты.

SELECT
  projects.title,
  categories.name AS category,
  COUNT(technologies.id) AS tech_count
FROM projects
JOIN categories ON categories.id = projects.category_id
JOIN project_technologies ON project_technologies.project_id = projects.id
JOIN technologies ON technologies.id = project_technologies.technology_id
GROUP BY projects.title, categories.name;
